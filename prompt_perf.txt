+static void qw3_count_layer_types_before(int n_layers,
+                                         int *n_full,
+                                         int *n_linear) {
+    if (n_layers < 0) n_layers = 0;

+    int full = 0;
+    int linear = 0;
+    for (int il = 0; il < n_layers; il++) {

+        else linear++;
